// USI LS conversion (agroponics style lossless like nom-o-matic)

{
    // LS recycler for 4 kerbals
	MODULE
	{
		name = USILS_LifeSupportRecyclerSwapOption
		ConverterName = Life Support Recycler
		StartActionName = Start Life Support
		StopActionName = Stop Life Support

		CrewCapacity = 4
		RecyclePercent = .5

		INPUT_RESOURCE
		{
			ResourceName = ElectricCharge
			Ratio = 1.0
		}
	}      
    
    
    // agroponics greenhouse
	MODULE
	{
		name = WBIModuleGreenhouse
		ConverterName = Agroponics
		StartActionName = Start Agroponics
		StopActionName = Stop Agroponics
		AutoShutdown = false
		GeneratesHeat = false
		UseSpecialistBonus = false
		SpecialistEfficiencyFactor = 1.0
		ExperienceEffect = ScienceSkill
		EfficiencyBonus = 1
		hoursPerCycle = 168 //Number of hours to grow crops
		minimumSuccess = 25
		criticalSuccess = 95
		criticalFail = 20
		cropResource = Supplies
        
		//Calibrated for 4 kerbals at 50% consumption for 28 days
		//Supports 2 kerbals
		cropYield = 560
        
		criticalSuccessBonus = 0.25
		failureLoss = 0.5
		repairSkill = ScienceSkill
		repairResource = MaterialKits
		repairAmount = 50

		criticalSuccessBonus = 0.25
		failureLoss = 0.25

		INPUT_RESOURCE
		{
			ResourceName = Mulch
			Ratio = 0.000842
		}
		INPUT_RESOURCE
		{
			ResourceName = Fertilizer
			Ratio = 0.000084
		}
		INPUT_RESOURCE
		{
			ResourceName = ElectricCharge
			Ratio = 6.4
		}

		//Calibrated for 4 kerbals at 5 supplies/day, after 28 days (168h)
		YIELD_RESOURCE
		{
			ResourceName = Supplies
			Ratio = 560
		}
	}


	RESOURCE
	{
		name = Mulch
		amount = 0
		maxAmount = 1000
		isTweakable = true
	}

	RESOURCE
	{
		name = Fertilizer
		amount = 50
		maxAmount = 50
		isTweakable = true
	}
    
    RESOURCE
	{
		name = Supplies
		amount = 0
		maxAmount = 1000
		isTweakable = true
	}
}
